```scss /// Map von z-index, sammelt alle Z-Ebenen der Applikation /// @access private /// @type Map /// @prop {String} key - Name der Ebene /// @prop {Number} value - Abbildung des Z-Wert eines Key's $z-indexes: ( 'modal': 5000, 'dropdown': 4000, 'default': 1, 'below': -1, ); /// Bekomm den z-index Wert durch den Ebenennamen /// @access public /// @param {String} $layer - Name der Ebene /// @return {Number} /// @require $z-indexes @function z($layer) { @if not map-has-key($z-indexes, $layer) { @error 'There is no layer named `#{$layer}` in $z-indexes.
