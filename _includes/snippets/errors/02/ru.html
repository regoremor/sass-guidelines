```scss /// Карта Z-index’ов, собирает все Z-слои приложения /// @access private /// @type Map /// @prop {String} key - Имя слоя /// @prop {Number} valye - значение Z, соответствующее ключу $z-indexes: ( 'modal': 5000, 'dropdown': 4000, 'default': 1, 'below': -1, ); /// Получение значения z-index из имени слоя /// @access public /// @param {String} $layer - Имя слоя /// @return {Number} /// @require $z-indexes @function z($layer) { @if not map-has-key($z-indexes, $layer) { @error 'Нет слоя с именем `#{$layer}` в $z-indexes. ' + 'Слой должен быть одним из #{map-keys($z-indexes)}.'; } @return map-get($z-indexes, $layer); } ```
